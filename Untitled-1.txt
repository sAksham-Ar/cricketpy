#include<bits/stdc++.h>

int min_dist(int a[], int b[],int n,int i,int check)
{
    if(check==n)
    return 0;
    else{
        int j,temp=INT_MAX;
        for(j=0;j<n;j++)
        {
            if(b[j]!=1 && j!=i)
            {
                b[j]=1;
                check++;
                temp=min(temp,dist(a[i],a[j])+min_dist(a,b,n,j));
                b[j]=0;
                check--;
            }
        }
        return temp;
    }
}
